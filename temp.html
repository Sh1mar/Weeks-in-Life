function getWeeksInMonth(year, month) { const firstDayOfMonth = new Date(year, month, 1); const lastDayOfMonth = new Date(year, month + 1, 0); const firstWeekDay = firstDayOfMonth.getDay(); // Sunday as 0, Monday as 1, etc. const lastDate = lastDayOfMonth.getDate(); // Calculate the number of days from the start of the month to the end of the first week const daysInFirstWeek = 7 - firstWeekDay; // Remaining days after the first week const remainingDays = lastDate - daysInFirstWeek; // Calculate full weeks in the remaining days const fullWeeks = Math.floor(remainingDays / 7); // Check if there are extra days that don't fit into a full week const extraDays = remainingDays % 7; // Total weeks is the first partial week + full weeks + possible extra week return 1 + fullWeeks + (extraDays > 0 ? 1 : 0); } function isLeapYear(year) { return (year % 4 === 0 && year % 100 != 0) || (year % 400 === 0); } function getWeeksinEachMonth(year){ const weeksInMonth = []; for(let months = 0; months < 12; months++){ weeksInMonth.push(getWeeksInMonth(year, months)); } return weeksInMonth; } function sum(accumulator,element) { return accumulator + element; } const currentYear = Number(new Date().getFullYear()); const weeksInMonths_arr = getWeeksinEachMonth(currentYear); console.log(weeksInMonths_arr); console.log(weeksInMonths_arr.reduce(sum));
